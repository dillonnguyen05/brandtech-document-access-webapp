rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function currentUserProfile() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }

    function isActiveAdmin() {
      return signedIn()
        && currentUserProfile().role == "admin"
        && currentUserProfile().status == "active";
    }

    function isActiveCustomer() {
      return signedIn()
        && currentUserProfile().role == "customer"
        && currentUserProfile().status == "active";
    }

    match /users/{userId} {
      allow read: if signedIn()
        && (request.auth.uid == userId || isActiveAdmin());

      allow create, update, delete: if false;
    }

    match /documents/{documentId} {
      allow read, write: if false;
    }

    match /accessRequests/{requestId} {
      allow read: if isActiveAdmin()
        || (
          isActiveCustomer()
          && resource.data.customerId == request.auth.uid
        );

      allow create, update, delete: if false;
    }

    match /auditLog/{auditId} {
      allow read, write: if false;
    }

    match /notifications/{notificationId} {
      allow read: if isActiveAdmin()
        || (
          isActiveCustomer()
          && resource.data.recipientId == request.auth.uid
        );

      allow create, update, delete: if false;
    }
  }
}

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function currentUserProfile() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }

    function isActiveAdmin() {
      return signedIn()
        && currentUserProfile().role == "admin"
        && currentUserProfile().status == "active";
    }

    function isActiveCustomer() {
      return signedIn()
        && currentUserProfile().role == "customer"
        && currentUserProfile().status == "active";
    }

    match /users/{userId} {
      allow create: if signedIn()
        && request.auth.uid == userId
        && request.resource.data.role == "customer"
        && request.resource.data.status == "pending";

      allow read: if signedIn()
        && (request.auth.uid == userId || isActiveAdmin());

      allow update: if isActiveAdmin()
        || (
          signedIn()
          && request.auth.uid == userId
          && request.resource.data.diff(resource.data).changedKeys().hasOnly([
            "emailVerified",
            "emailVerifiedAt"
          ])
        );

      allow delete: if isActiveAdmin();
    }

    match /documents/{documentId} {
      allow read: if isActiveAdmin() || isActiveCustomer();
      allow write: if isActiveAdmin();
    }

    match /accessRequests/{requestId} {
      allow create: if isActiveCustomer()
        && request.resource.data.customerId == request.auth.uid
        && request.resource.data.status == "pending";

      allow read: if isActiveAdmin()
        || (
          isActiveCustomer()
          && resource.data.customerId == request.auth.uid
        );

      allow update: if isActiveAdmin()
        || (
          isActiveCustomer()
          && resource.data.customerId == request.auth.uid
          && resource.data.status == "denied"
          && request.resource.data.customerId == request.auth.uid
          && request.resource.data.status == "pending"
        );

      allow delete: if isActiveAdmin();
    }

    match /auditLog/{auditId} {
      allow read, write: if isActiveAdmin();
    }

    match /notifications/{notificationId} {
      allow create: if isActiveAdmin();

      allow read: if isActiveAdmin()
        || (
          isActiveCustomer()
          && resource.data.recipientId == request.auth.uid
        );

      allow update: if isActiveAdmin()
        || (
          isActiveCustomer()
          && resource.data.recipientId == request.auth.uid
          && request.resource.data.diff(resource.data).changedKeys().hasOnly([
            "read",
            "readAt"
          ])
        );

      allow delete: if isActiveAdmin();
    }
  }
}
